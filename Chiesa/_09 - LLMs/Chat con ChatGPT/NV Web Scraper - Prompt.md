async def file_hash(path):
    """Calcola hash SHA256 di un file locale (se esiste)."""
    h = hashlib.sha256()
    try:
        async with aiofiles.open(path, 'rb') as f:

                chunk = await f.read(4096)
                if not chunk:

                h.update(chunk)
        return h.hexdigest()
    except FileNotFoundError:
        return None
